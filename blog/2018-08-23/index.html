class method, static method and object method
ad
Aug 23, 2018
One minute read
class A(object):
    def foo(self,x):
        # executing foo(<__main__.A object at 0xb7dbef0c>,1)
        print "executing foo(%s,%s)"%(self,x)

    @classmethod
    def class_foo(cls,x):
        # executing class_foo(<class '__main__.A'>,1)
        print "executing class_foo(%s,%s)"%(cls,x)

    @staticmethod
    def static_foo(x):
        # executing static_foo(1)
        print "executing static_foo(%s)"%x    

a=A()
  • You can call class_foo method by a.class_foo or A.class_foo.
  • You can call static_foo method by a.static_foo or A.static_foo.
